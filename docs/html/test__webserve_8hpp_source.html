148 void multi_plexing_run_with_epoll()
149 {
150 int kq, event_num;
151 std::vector<struct kevent> change_list; // kqueue에 등록할 event list
152 struct kevent event_list[10]; // event 리스트
153 struct kevent event_temp; // event 임시 변수
154 struct kevent* curr_event;
155 int clnt_sock;
156 int str_len;
157 char message[BUFSIZE];
158
159 // kqueue 초기화
160 if ((kq = kqueue()) == -1)
161 throw std::exception();
162
163 // 서버 리스닝 소켓 change list에 추가
164 EV_SET(&event_temp, socket_fd, EVFILT_READ, EV_ADD | EV_ENABLE, 0, 0, NULL);
165 change_list.push_back(event_temp);
166
167 while (1)
168 {
169 // kevent() change list 안에 이벤트들을 새로 등록하고, 이벤트 발생 감지
170 // -1 : 오류 발생
171 // 0 : 타임 아웃
172 // 1 이상 : 이벤트 발생한 개수
173 event_num = kevent(kq, &change_list[0], change_list.size(), event_list, 10, NULL);
174 if (event_num == -1)
175 break;
176 change_list.clear(); // change list 초기화
177 for (int i = 0; i < event_num; i++)
178 {
179 curr_event = &event_list[i];
180 // 현재 이벤트가 에러 발생
181 if (curr_event->flags & EV_ERROR)
182 {
183 if (curr_event->ident == socket_fd)
184 std::exception();
185 else
186 {
187 close(curr_event->ident);
188 }
189 }
190 // 현재 이벤트가 read일 경우
191 else if (curr_event->filter == EVFILT_READ)
192 {
193 // 서버 리스닝 소켓에서 이벤트 발생
194 if (curr_event->ident == socket_fd)
195 {
196 // client accpet
197 clnt_sock = accept_server();
198 // 클라이언트 소켓 change list에 추가
199 EV_SET(&event_temp, clnt_sock, EVFILT_READ, EV_ADD | EV_ENABLE, 0, 0, NULL);
200 change_list.push_back(event_temp);
201 EV_SET(&event_temp, clnt_sock, EVFILT_WRITE, EV_ADD | EV_ENABLE, 0, 0, NULL);
202 change_list.push_back(event_temp);
203 }
204 // 클라이언트 소켓에서 이벤트 발생
205 else
206 {
207 // read 에러이거나 클라이언트 disconnect 요청
208 if ((str_len = read(curr_event->ident, message, BUFSIZE)) == 0)
209 {
210 close(curr_event->ident);
211 }
212 // read data
213 else
214 {
215 request_handler(&curr_event->ident, str_len, message);
216 }
217 }
218 }
219 }
220 }
221 close(socket_fd);
222 }
